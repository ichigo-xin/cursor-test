# 网站目录结构

├── public/                 # 静态资源
├── src/                    # 源代码
│   ├── components/        # 公共组件
│   │   ├── Header/       # 页头组件
│   │   ├── Footer/       # 页脚组件
│   │   └── Upload/       # 上传组件
│   │
│   ├── pages/            # 页面组件
│   │   ├── Home/        # 首页
│   │   ├── FileConvert/ # 文件转换
│   │   ├── ImageCompress/# 图片压缩
│   │   ├── ImageCrop/   # 图片裁剪
│   │   ├── ImageFormat/ # 图片格式转换
│   │   └── ImageWatermark/# 图片水印
│   │
│   ├── services/         # API 服务
│   │   ├── fileService.js
│   │   └── imageService.js
│   │
│   ├── utils/           # 工具函数
│   │   ├── compress.js  # 压缩工具
│   │   ├── format.js    # 格式转换
│   │   └── watermark.js # 水印工具
│   │
│   ├── styles/          # 样式文件
│   │   ├── global.css
│   │   └── variables.css
│   │
│   ├── App.js           # 应用入口
│   └── index.js         # 项目入口
│
├── server/               # 后端代码（如需要）
│   ├── controllers/     # 控制器
│   ├── routes/         # 路由
│   ├── services/       # 服务
│   └── index.js        # 服务入口
│
├── package.json
└── README.md
